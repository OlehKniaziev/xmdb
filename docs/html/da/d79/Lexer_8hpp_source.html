65static inline SourceLocation locate_token(StringView source, Token token) {
66 OK_ASSERT((uintptr_t) token.data.data >= (uintptr_t) source.data);
67
68 ptrdiff_t token_offset = token.data.data - source.data;
69
70 uint32_t line = 1;
71 uint32_t column = 1;
72 for (ptrdiff_t i = 0; i < token_offset; ++i) {
73 // TODO: support DOS-style newlines
74 if (source[i] == '\n') {
75 line++;
76 column = 1;
77 } else {
78 column++;
79 }
80 }
81
82 return SourceLocation{
83 .line = line,
84 .column = column,
85 .length = (uint32_t) token.data.count,
86 };
87}
